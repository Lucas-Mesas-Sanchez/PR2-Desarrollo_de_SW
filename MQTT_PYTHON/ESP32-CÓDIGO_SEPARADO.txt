#include <Arduino.h>
#include <WiFi.h>
#include <PubSubClient.h>

#include "Config.h"


//---------------------------------------------
// setup.ino
//---------------------------------------------
extern WiFiClient espClient;
extern PubSubClient client;

extern const int pinSensor;
extern const int ledTemp;
extern const int btnRellenar;
extern const int btnVaciar;
extern const int btnEmergencia;
extern const int ledFullNuevas;
extern const int ledFullMalas;

void setup() {
  Serial.begin(BAUDS);

  pinMode(pinSensor, INPUT);
  pinMode(ledTemp, OUTPUT);
  pinMode(ledFullNuevas, OUTPUT);
  pinMode(ledFullMalas, OUTPUT);

  pinMode(btnRellenar, INPUT_PULLUP);
  pinMode(btnVaciar, INPUT_PULLUP);
  pinMode(btnEmergencia, INPUT_PULLUP);

  digitalWrite(ledTemp, LOW);
  digitalWrite(ledFullNuevas, LOW);
  digitalWrite(ledFullMalas, LOW);

  setupWiFi();
  setupMQTT();
}


//---------------------------------------------
// loop.ino
//---------------------------------------------
extern PubSubClient client;

void loop() {

  if (!client.connected()) {
    reconnectMQTT();
  }
  client.loop();

  comprobarTemperatura();
  comprobarLogistica();
  comprobarEmergencia();

  delay(100);
}


//---------------------------------------------
// comunicaciones.ino
//---------------------------------------------
#include "Config.h"

WiFiClient espClient;
PubSubClient client(espClient);

void callback(char* topic, byte* payload, unsigned int length) {
  String mensaje = "";
  for (unsigned int i = 0; i < length; i++) {
    mensaje += (char)payload[i];
  }

  if (mensaje == "AVISO_MALAS_LLENA") {
    digitalWrite(19, HIGH);
  }
  else if (mensaje == "AVISO_NUEVAS_LLENA") {
    digitalWrite(18, HIGH);
  }
}

void setupWiFi() {
  WiFi.begin(NET_SSID, NET_PASSWD);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
  }
}

void setupMQTT() {
  client.setServer(MQTT_SERVER_IP, MQTT_SERVER_PORT);
  client.setCallback(callback);
}

void reconnectMQTT() {
  while (!client.connected()) {
    String clientId = "ESP32_PR2_";
    clientId += DEVICE_GIIROB_PR2_ID;

    if (client.connect(clientId.c_str(), MQTT_USERNAME, MQTT_PASSWORD)) {
      client.subscribe("mqtt/pr2/datacenter/g2");
    } else {
      delay(5000);
    }
  }
}


//---------------------------------------------
// funciones.ino
//---------------------------------------------
#include "Config.h"

const int pinSensor     = 34;
const int ledTemp       = 2;
const int btnRellenar   = 4;
const int btnVaciar     = 5;
const int btnEmergencia = 13;
const int ledFullNuevas = 18;
const int ledFullMalas  = 19;

bool ultimoEstadoBtnRellenar = HIGH;
bool ultimoEstadoBtnVaciar   = HIGH;
bool ultimoEstadoEmergencia  = HIGH;

extern PubSubClient client;

void comprobarTemperatura() {
  int lectura = analogRead(pinSensor);

  if (lectura > 2500) {
    digitalWrite(ledTemp, HIGH);
    client.publish("mqtt/pr2/datacenter/g2", "TARJETA_DEFECTUOSA");
    delay(2000);
  } else {
    digitalWrite(ledTemp, LOW);
  }
}

void comprobarLogistica() {
  bool estadoRellenar = digitalRead(btnRellenar);
  bool estadoVaciar   = digitalRead(btnVaciar);

  if (estadoRellenar == LOW && ultimoEstadoBtnRellenar == HIGH) {
    client.publish("mqtt/pr2/datacenter/g2", "RELLENADO_TARJETAS");
    digitalWrite(ledFullNuevas, LOW);
    delay(200);
  }

  if (estadoVaciar == LOW && ultimoEstadoBtnVaciar == HIGH) {
    client.publish("mqtt/pr2/datacenter/g2", "VACIADO_TARJETAS_DEFECTUOSAS");
    digitalWrite(ledFullMalas, LOW);
    delay(200);
  }

  ultimoEstadoBtnRellenar = estadoRellenar;
  ultimoEstadoBtnVaciar   = estadoVaciar;
}

void comprobarEmergencia() {
  bool estado = digitalRead(btnEmergencia);

  if (estado == LOW && ultimoEstadoEmergencia == HIGH) {
    client.publish("mqtt/pr2/datacenter/g2", "EMERGENCIA_STOP");
    digitalWrite(ledTemp, HIGH);
  }

  ultimoEstadoEmergencia = estado;
}
